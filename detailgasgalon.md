        function toggleWarning(show) {
            const warningBox = document.getElementById('warning-box');
            if(show) {
                warningBox.style.display = 'flex';
                warningBox.style.opacity = '1';
            } else {
                warningBox.style.opacity = '0';
                setTimeout(() => { warningBox.style.display = 'none'; }, 300);
            }

            // Update border styles for radio selection
            const labels = document.querySelectorAll('label[onclick]');
            labels.forEach(label => {
                const input = label.querySelector('input');
                if(input.checked) {
                    label.classList.add('border-primary-container', 'bg-surface-container-low');
                    label.classList.remove('border-outline-variant', 'hover:bg-surface-container-low');
                } else {
                    label.classList.remove('border-primary-container', 'bg-surface-container-low');
                    label.classList.add('border-outline-variant', 'hover:bg-surface-container-low');
                }
            });
        }
